document.addEventListener('DOMContentLoaded', function() {
  // Initialize all skill category cards as collapsed
  const skillCards = document.querySelectorAll('.skill-category-card');
  
  skillCards.forEach(card => {
    const header = card.querySelector('.skill-card-header');
    const content = card.querySelector('.skill-card-content');
    const toggle = card.querySelector('.skill-card-toggle');
    
    // Add click handler to the header
    header.addEventListener('click', function() {
      const isExpanded = card.classList.contains('expanded');
      
      if (isExpanded) {
        // Collapse
        card.classList.remove('expanded');
        content.style.display = 'none';
        toggle.style.transform = 'rotate(0deg)';
      } else {
        // Expand
        card.classList.add('expanded');
        content.style.display = 'block';
        toggle.style.transform = 'rotate(180deg)';
        
        // Add entrance animation
        content.style.opacity = '0';
        content.style.transform = 'translateY(-10px)';
        
        // Trigger animation
        setTimeout(() => {
          content.style.transition = 'all 0.3s cubic-bezier(0.25, 0.46, 0.45, 0.94)';
          content.style.opacity = '1';
          content.style.transform = 'translateY(0)';
        }, 10);
      }
    });
    
    // Add keyboard support
    header.addEventListener('keydown', function(e) {
      if (e.key === 'Enter' || e.key === ' ') {
        e.preventDefault();
        header.click();
      }
    });
    
    // Make header focusable
    header.setAttribute('tabindex', '0');
    header.setAttribute('role', 'button');
    header.setAttribute('aria-expanded', 'false');
    
    // Update aria-expanded when card state changes
    const observer = new MutationObserver(function(mutations) {
      mutations.forEach(function(mutation) {
        if (mutation.attributeName === 'class') {
          const isExpanded = card.classList.contains('expanded');
          header.setAttribute('aria-expanded', isExpanded.toString());
        }
      });
    });
    
    observer.observe(card, { attributes: true, attributeFilter: ['class'] });
  });
});
</script>
